\qns{(PRACTICE) To Invert or Not to Invert? That is the Question...}

For each of the following cases, determine if $\textbf{A}\in\mathbb{R}^{n\times n}$ is invertible, not invertible, or you don't have enough information.
\sol{
	It may be useful to go over portions of the invertible matrix theorem that the students have seen.}
\begin{enumerate}
\qitem\label{ques:nontrivialNullspace}{
	$\textbf{A}\vec{x} = \vec{0}$ for some $\vec{x} \neq \vec{0}$
}
\sol{}
\ans{
	Not invertible.
}

\qitem\label{ques:linIndepCols}{
	$\textbf{A}$ has linearly independent columns
}
\sol{}
\ans{
	Invertible.
}

\qitem\label{ques:linIndepRows}{
	$\textbf{A}$ has linearly independent rows
}
\sol{}
\ans{
	Invertible.
}

\qitem\label{ques:matMul}{
	$\textbf{BA}$ is invertible
}
\sol{
	You cannot assume that $\textbf{BA}^{-1} = A^{-1}B^{-1}$ when solving this!
}
\ans{
	Invertible.

	\begin{align*}
		\textbf{BA} \text{ invertible} \Rightarrow \exists \textbf{C} \text{ such that } \textbf{C}(\textbf{BA}) &= \textbf{I}\\
		\textbf{I} &= (\textbf{CB})\textbf{A}\\
	\end{align*}
	and so \textbf{A} is invertible with an inverse \textbf{CB}
}
\end{enumerate}
